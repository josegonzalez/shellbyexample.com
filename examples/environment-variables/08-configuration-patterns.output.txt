Without set -a:
  Parent sees APP_HOST: 0.0.0.0
  Child sees APP_HOST:  <not visible>

  Parent sees APP_HOST: 0.0.0.0
  Child sees APP_HOST:  0.0.0.0

  APP_PORT=3000
  APP_HOST=0.0.0.0
